 *   <li>El ISBN debe ser √∫nico en el sistema</li>
 *   <li>El precio base debe ser mayor que 0</li>

 * book.setBasePrice(new BigDecimal("45.99"));

@Entity
@Table(name = "books")
@Data
@NoArgsConstructor

    /**
     * Identificador √∫nico del libro en el sistema.
     * 
     * <p>Generado autom√°ticamente por la base de datos usando estrategia
     * de identidad. Este ID se usa como clave primaria y para todas las
     * referencias a este libro desde otras entidades.

    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;
    
    /**

     * La longitud m√°xima permitida es de 255 caracteres para compatibilidad

    @NotBlank(message = "El t√≠tulo del libro es obligatorio")

    /**

    /**

    @DecimalMin(value = "0.01", message = "El precio debe ser mayor que 0")

    @Column(name = "base_price", precision = 8, scale = 2, nullable = false)
    private BigDecimal basePrice;
    
    /**

    /**

    @ManyToMany(fetch = FetchType.LAZY, cascade = {CascadeType.PERSIST, CascadeType.MERGE})

        name = "book_authors",

        inverseJoinColumns = @JoinColumn(name = "author_id")

    private Set<Author> authors;
    
    /**

    /**

    /**

    /**

     * <p>Actualiza autom√°ticamente el timestamp de √∫ltima modificaci√≥n.
     * Este m√©todo es llamado por JPA antes de realizar el UPDATE en base de datos.

    }
    
    /**
     * Verifica si el libro est√° disponible para venta.
     * 
     * <p>Un libro est√° disponible si tiene stock positivo. Este m√©todo

    }
    
    /**
     * Reduce el stock del libro en la cantidad especificada.
     * 
     * <p>Valida que hay suficiente stock antes de realizar la reducci√≥n.
     * Este m√©todo debe usarse dentro de transacciones para garantizar
     * consistencia en operaciones de venta.

    public void reduceStock(Integer quantity) {
        if (quantity <= 0) {
            throw new IllegalArgumentException("La cantidad debe ser positiva");
        }
        
        if (stockQuantity < quantity) {
            throw new InsufficientStockException(
                String.format("Stock insuficiente. Disponible: %d, Solicitado: %d", 
                             stockQuantity, quantity)
            );
